Usage

This project is mainly dedicated to the people who do not have as good as eyeseight as we do,

Working

When the Ultrasonic sensor senses the object by the reflection of Ultrasonic waves it sends the current to the arduino which then strats beeping and the buzzer starts vibrationg alerting the person to swich directions.

Code


#define trigPin 2 // Ultrasonic sensor TRIG pin

#define echoPin 3 // Ultrasonic sensor ECHO pin

#define buzzerPin 4 // Buzzer pin


long duration;

int distance;


void setup() {

pinMode(trigPin, OUTPUT);

pinMode(echoPin, INPUT);

pinMode(buzzerPin, OUTPUT);

Serial.begin(9600);

}


void loop() {

// Trigger ultrasonic pulse

digitalWrite(trigPin, LOW);

delayMicroseconds(2);

digitalWrite(trigPin, HIGH);

delayMicroseconds(10);

digitalWrite(trigPin, LOW);


// Read echo

duration = pulseIn(echoPin, HIGH);

distance = duration * 0.034 / 2; // Convert to cm


Serial.print("Distance: ");

Serial.print(distance);

Serial.println(" cm");


// Obstacle detection with buzzer intensity control

if (distance > 0 && distance <= 250) {

// The closer the obstacle, the faster the buzzer

int delayTime = map(distance, 1, 50, 170, 400); // 1cm=very fast, 20cm=slow

digitalWrite(buzzerPin, HIGH);

delay(50);

digitalWrite(buzzerPin, LOW);

delay(delayTime);

} else {

digitalWrite(buzzerPin, LOW); // No obstacle = silent

}

}
